from flask import Flask, jsonify
from flask_cors import CORS

# Initialize the Flask application
app = Flask(__name__)
CORS(app)

# --- Mock Data ---
def get_dashboard_data():
    """Returns a dictionary of mock data for the dashboard."""
    return {
        "kpi": {
            "total_revenue": {"value": 45231.89, "change": 20.1},
            "subscriptions": {"value": 2350, "change": 180.1},
            "sales": {"value": 12234, "change": -2.5},
            "active_now": {"value": 573, "change_label": "+201 since last hour"}
        },
        "revenue_overview": [
            { "name": 'Jan', "revenue": 4000, "profit": 2400 },
            { "name": 'Feb', "revenue": 3000, "profit": 1398 },
            { "name": 'Mar', "revenue": 5000, "profit": 9800 },
        ],
        # ... more data ...
    }

@app.route('/api/data')
def data_endpoint():
    """Serves the dashboard data in JSON format."""
    data = get_dashboard_data()
    return jsonify(data)

if __name__ == '__main__':
    app.run(debug=True, port=5000)
